    participant User as User (Touch/Mouse)
    participant Joystick as UI_Joystick (VirtualJoystick)
    participant PlayerMove as PlayerMovement (Script)
    participant CC as CharacterController (Component)
    participant Physics as Unity_Physics_Engine

    Note over User, Physics: Player Movement Flow (Per Frame)

    User->>Joystick: Input Drag (Position)
    activate Joystick
    Joystick->>Joystick: Calculate Input Vector (Normalize -1 to 1)
    Joystick-->>PlayerMove: GetInput() / Send InputVector (Vector2)
    deactivate Joystick

    activate PlayerMove
    PlayerMove->>PlayerMove: Check MoveState (Boolean)
    
    alt If MoveState is TRUE
        PlayerMove->>PlayerMove: Convert Vector2 to Vector3 (X, 0, Z)
        PlayerMove->>PlayerMove: Multiply by Speed & DeltaTime
        PlayerMove->>CC: Move(Direction)
        activate CC
        CC->>Physics: Update Position & Collision
        Physics-->>CC: Resolve Collisions
        CC-->>PlayerMove: Final Position
        deactivate CC
        PlayerMove->>PlayerMove: Set isWalking (True)
    else If MoveState is FALSE
        PlayerMove->>PlayerMove: Skip Movement Logic
        PlayerMove->>PlayerMove: Set isWalking (False)
    end
    
    PlayerMove->>CC: Apply Gravity (Vertical Velocity)
    activate CC
    CC->>Physics: Apply Y Movement
    deactivate CC
    
    deactivate PlayerMove
